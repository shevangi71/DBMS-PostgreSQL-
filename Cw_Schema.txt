CREATE TABLE event (    ecode        CHAR(4),    edesc        VARCHAR(20),    elocation    VARCHAR(20),    edate        DATE,    etime        TIME,    emax         SMALLINT );CREATE TABLE spectator (    sno          INTEGER,    sname        VARCHAR(20),    semail       VARCHAR(20) );CREATE TABLE ticket (    tno          INTEGER,    ecode        CHAR(4),    sno          INTEGER );CREATE TABLE cancel (    tno          INTEGER,    ecode        CHAR(4),    sno          INTEGER,    cdate        TIMESTAMP,    cuser        VARCHAR(128) );
